import Link from 'next/link';
import { MessageSquare, Users } from 'lucide-react';
import { Card, CardContent } from '@/components/ui/card'; // Adjust path based on your setup

// Define category interface
interface Category {
  id: string | number;
  slug: string;
  name: string;
  description?: string;
  icon?: string;
  postCount?: number;
  color?: string;
}

// Define icon and image mappings
const iconMap: { [key: string]: React.ComponentType } = {
  message: MessageSquare,
  users: Users,
};

const categoryImages: { [key: string]: string } = {
  'category-slug-1': '/images/category1.jpg',
  'category-slug-2': '/images/category2.jpg',
};

// Component (assuming categories is passed as a prop)
interface Props {
  categories?: Category[];
}

export default function CategoryGrid({ categories }: Props) {
  return (
    <>
      {categories && (
        <div className="grid grid-cols-1 md:grid-cols-3 gap-6 max-w-6xl mx-auto">
          {categories.slice(0, 6).map((category: Category) => {
            const IconComponent = iconMap[category.icon || 'message'] || MessageSquare;
            const postCount = category.postCount ?? 50;
            const memberCount = category.memberCount ?? 20;

            return (
              <Link key={category.id} href={`/category/${category.slug}`}>
                <Card className="hover:shadow-md transition-shadow cursor-pointer h-full">
                  <CardContent className="p-6">
                    <div className="flex items-center space-x-3 mb-4">
                      <div className="w-12 h-12 rounded-xl overflow-hidden">
                        <img
                          src={categoryImages[category.slug] || '/images/default.jpg'}
                          alt={category.name}
                          className="w-full h-full object-cover"
                        />
                      </div>
                      <div className="flex-1">
                        <h3 className="font-semibold text-gray-900 text-sm leading-tight">
                          {category.name}
                        </h3>
                      </div>
                    </div>

                    <p className="text-sm text-gray-600 mb-4 leading-relaxed">
                      {category.description}
                    </p>

                    <div className="space-y-2">
                      <div className="flex items-center justify-between text-xs text-gray-500">
                        <div className="flex items-center space-x-1">
                          <MessageSquare className="w-3 h-3" />
                          <span>{postCount} posts</span>
                        </div>
                        <div className="flex items-center space-x-1">
                          <Users className="w-3 h-3" />
                          <span>{memberCount}</span>
                        </div>
                      </div>

                      <div className="flex w-full rounded-lg overflow-hidden border border-gray-200">
                        <Link href="/api/login" className="flex-1">
                          <button
                            className="w-full text-white text-xs font-semibold py-2 hover:opacity-90 transition-opacity"
                            style={{ backgroundColor: category.color || '#3b82f6' }}
                          >
                            Join
                          </button>
                        </Link>
                        <div className="w-px bg-white/30"></div>
                        <button
                          className="flex-1 text-white text-xs font-semibold py-2 hover:opacity-90 transition-opacity"
                          style={{ backgroundColor: category.color || '#3b82f6' }}
                        >
                          Explore
                        </button>
                      </div>
                    </div>
                  </CardContent>
                </Card>
              </Link>
            );
          })}
        </div>
      )}
    </>
  );
}
